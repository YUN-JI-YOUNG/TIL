### 1859. 백만장자 프로젝트

```python
T = int(input())
for test in range(1,T+1):
    N = int(input())
    list_1 = list(map(int, input().split(" ")))
    max_i = list_1[-1]
    result = 0
    for i in range(N-2, -1, -1):
        if list_1[i] <= max_i:
            result += max_i - list_1[i]
        elif list_1[i] > max_i:
            max_i = list_1[i]        
    print('#{0} {1}'.format(test,result))
```

### 1923. 간단한 369게임    
```python
n = int(input())

nlist = list(range(1,n+1))
for i in range(0,n):
    count = 0
    s = str(nlist[i])
    for j in range(len(s)):
        if ('3' in s[j]) or ('6' in s[j]) or ('9' in s[j]):
            count += 1
            num = '-' * count
        elif ('3' not in s) and ('6' not in s) and ('9' not in s):
            num = s

    print(num, end = ' ')
```

### 2007. 패턴 마디의 길이   
```python
T = int(input())

for n in range(1,T+1):
    slist = (input().split(' '))[0]
    result = ''
    for i in range(1, len(slist)):
        if slist[0] == slist[i]:
            if slist[0:i] == slist[i:2*i]:
                result = slist[0:i]
                print('#{0} {1}'.format(n, len(result)))
                break
```

### 4880. 토너먼트 카드게임   
```python   
T = int(input())

def match(l,r):
    if cards[l] - cards[r] == -2 or cards[l] - cards[r] == 1 or cards[l] - cards[r] == 0:
        return l
    else:
        return r

def half(s,e):
    if s == e:   # s와 e가 같다는건 번호가 같은거니까 동일한 사람 지칭
        return s
    m = (s+e)//2
    left = half(s, m)
    right = half(m+1, e)
    return match(left, right)


for tc in range(1,T+1):
    N = int(input())
    cards = list(map(int,input().split()))

    print('#{} {}'.format(tc, half(0,N-1)+1)) 
```
